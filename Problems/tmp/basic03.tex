\question Given \mbox{$V_{1} = 2.34$} and \mbox{$R_{1} = \mbox{$6 \units{k \Omega}$}$}, find the current  $I_R = V_1/R_1$


\begin{solutions}
\textbf{Solution}

\mbox{$I_{R}=V_{1}/R_{1}  = (2.34)/(6e3) = 390e-6$}\\
\hlite{\mbox{$I_{R} = 390e-6$}}
\end{solutions}
